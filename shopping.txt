    <script>
        /* --- UPDATED DATA WITH YOUR UPLOADED FILENAMES --- */
        const inventory = [
            { id: 'dress', price: 300, plural: false, img: 'dress.png' }, // Using standard names, update if different
            { id: 'skirt', price: 200, plural: false, img: 'skirt.png' },
            { id: 'sweater', price: 400, plural: false, img: 'sweater.png' },
            { id: 'T-shirt', price: 200, plural: false, img: 't-shirt.png' },
            { id: 'pants', price: 200, plural: true, img: 'pants.png' },
            { id: 'shorts', price: 100, plural: true, img: 'shorts.png' },
            { id: 'sneakers', price: 600, plural: true, img: 'sneakers.jpg' },
            { id: 'socks', price: 100, plural: true, img: 'socks.png' },
            { id: 'coat', price: 700, plural: false, img: 'coat.jpg' },
            { id: 'vest', price: 300, plural: false, img: 'vest.png' },
            { id: 'scarf', price: 200, plural: false, img: 'scarf.png' },
            { id: 'gloves', price: 200, plural: true, img: 'gloves.jpg' },
            { id: 'sandals', price: 400, plural: true, img: 'sandals.png' }
        ];

        let balance = 0;
        let cart = {};
        let unlockedItems = new Set();
        let currentTyped = "";

        function startGame() {
            document.getElementById('start-screen').style.display = 'none';
            document.getElementById('game-container').style.display = 'grid';
            initShop();
            initKeyboard();
        }

        function initShop() {
            const grid = document.getElementById('shop-grid');
            grid.innerHTML = '';
            inventory.forEach(item => {
                const div = document.createElement('div');
                div.className = `item-card ${unlockedItems.has(item.id) ? 'unlocked' : ''}`;
                div.id = `card-${item.id}`;
                div.onclick = () => addToCart(item.id);
                div.innerHTML = `
                    <img src="${item.img}" class="item-img" onerror="this.src='https://via.placeholder.com/100?text=${item.id}'">
                    <div style="text-transform: capitalize; font-weight:bold; font-size:1.1rem">${item.id}</div>
                    <div class="item-price">$${item.price}</div>
                `;
                grid.appendChild(div);
            });
        }

        function addToCart(id) {
            cart[id] = (cart[id] || 0) + 1;
            updateCartUI();
        }

        function updateCartUI() {
            const list = document.getElementById('cart-list');
            let html = '';
            let total = 0;
            for (let id in cart) {
                const item = inventory.find(i => i.id === id);
                html += `<div style="margin-bottom:8px">🛒 ${id} x${cart[id]} ($${item.price * cart[id]})</div>`;
                total += item.price * cart[id];
            }
            list.innerHTML = html || 'The bag is empty!';
            document.getElementById('total-cost').innerText = total;
        }

        /* --- KEYBOARD LOGIC --- */
        function initKeyboard() {
            const grid = document.getElementById('keys-grid');
            "ABCDEFGHIJKLMNOPQRSTUVWXYZ-".split('').forEach(char => {
                const k = document.createElement('div');
                k.className = 'key';
                k.innerText = char;
                k.onclick = () => { currentTyped += char; updateKBScreen(); };
                grid.appendChild(k);
            });
            const del = document.createElement('div');
            del.className = 'key'; del.innerText = 'DEL'; del.style.background = "#ff6b6b"; del.style.color = "white";
            del.onclick = () => { currentTyped = currentTyped.slice(0,-1); updateKBScreen(); };
            grid.appendChild(del);
        }

        function updateKBScreen() {
            document.getElementById('kb-screen').innerText = currentTyped || "...";
        }

        function checkSpelling() {
            const found = inventory.find(i => i.id.toUpperCase() === currentTyped.toUpperCase());
            if (found) {
                unlockedItems.add(found.id);
                initShop();
                const card = document.getElementById(`card-${found.id}`);
                card.classList.add('glow');
                currentTyped = "";
                updateKBScreen();
                alert(`🔓 You unlocked the ${found.id.toUpperCase()}!`);
            } else {
                alert("Incorrect spelling! Try again.");
                currentTyped = "";
                updateKBScreen();
            }
        }

        /* --- PAYMENT & GRAMMAR --- */
        function openPayment() {
            const total = parseInt(document.getElementById('total-cost').innerText);
            if (total === 0) return alert("Your bag is empty!");
            if (balance < total) return alert("You need more money! Click 'Earn Money'.");

            const modal = document.getElementById('modal-overlay');
            const body = document.getElementById('modal-body');
            modal.style.display = 'flex';
            
            let questionsHTML = '<h2 style="color:var(--primary)">📝 ESL Grammar Challenge</h2><p>Finish these sentences to pay:</p>';
            for (let id in cart) {
                const item = inventory.find(i => i.id === id);
                // Grammar Logic: Pants and Shorts are ALWAYS plural
                const isPlural = item.plural || cart[id] > 1;
                const qText = isPlural ? `How much are the ${id}?` : `How much is the ${id}?`;
                const expected = isPlural ? `They're ${item.price * cart[id]} dollars` : `It's ${item.price * cart[id]} dollars`;
                
                questionsHTML += `
                    <div class="grammar-q">
                        <p><strong>${qText}</strong></p>
                        <input type="text" class="g-input" data-ans="${expected.toLowerCase()}" placeholder="Type: ${expected}">
                    </div>
                `;
            }
            questionsHTML += `<button class="btn-ui btn-pay" onclick="validateGrammar()">COMPLETE PURCHASE</button>`;
            body.innerHTML = questionsHTML;
        }

        function validateGrammar() {
            const inputs = document.querySelectorAll('.g-input');
            let allCorrect = true;
            inputs.forEach(input => {
                const userVal = input.value.toLowerCase().trim().replace(/[.,!]/g, "");
                const correctVal = input.getAttribute('data-ans').toLowerCase().trim().replace(/[.,!]/g, "");
                
                if (userVal !== correctVal) {
                    input.style.borderColor = "red";
                    allCorrect = false;
                } else {
                    input.style.borderColor = "green";
                }
            });

            if (allCorrect) {
                showReceipt();
            } else {
                alert("Check your grammar! Remember: It's ... dollars or They're ... dollars.");
            }
        }

        function showReceipt() {
            const total = parseInt(document.getElementById('total-cost').innerText);
            const oldBalance = balance;
            balance -= total;
            document.getElementById('balance').innerText = balance;

            let itemsList = '';
            for (let id in cart) {
                itemsList += `<p>${id.toUpperCase()} x${cart[id]} ............ $${inventory.find(i=>i.id===id).price * cart[id]}</p>`;
            }

            const body = document.getElementById('modal-body');
            body.innerHTML = `
                <div class="receipt">
                    <center><h2>🛍️ TEACHER KHEN MALL</h2>
                    <p>Official Receipt</p></center>
                    <hr>
                    ${itemsList}
                    <hr>
                    <p><strong>GRAND TOTAL: $${total}</strong></p>
                    <p>Balance: $${oldBalance}</p>
                    <p>Change: $${balance}</p>
                    <hr>
                    <center><p>Great job with your grammar!<br>See you next time!</p></center>
                </div>
                <button class="btn-ui btn-kb" style="width:100%; margin-top:20px; font-size:1.2rem" onclick="location.reload()">PLAY AGAIN</button>
            `;
        }

        /* --- WHEEL & DRAG --- */
        function spinWheel() {
            const wheel = document.getElementById('wheel-container');
            const deg = Math.floor(Math.random() * 3600) + 720;
            wheel.style.transform = `rotate(${deg}deg)`;
        }

        function togglePanel(id) {
            const p = document.getElementById(id);
            p.style.display = (p.style.display === 'block') ? 'none' : 'block';
        }

        function earnMoney() {
            balance += 200;
            document.getElementById('balance').innerText = balance;
            alert("💰 You completed a quiz and earned $200!");
        }

        /* DRAG LOGIC */
        let activeEl = null, offX = 0, offY = 0;
        function dragStart(e, id) {
            activeEl = document.getElementById(id);
            offX = activeEl.offsetLeft - e.clientX;
            offY = activeEl.offsetTop - e.clientY;
            document.onmousemove = (e) => {
                activeEl.style.left = (e.clientX + offX) + 'px';
                activeEl.style.top = (e.clientY + offY) + 'px';
            };
            document.onmouseup = () => { document.onmousemove = null; activeEl = null; };
        }
    </script>
